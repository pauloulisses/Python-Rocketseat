# LISTAS E TUPLAS

# LISTA (list)
# Mutável (pode alterar, adicionar e remover itens)
# Usa colchetes → [ ]
# Mais usada para dados que mudam





# TUPLA (tuple)

# Imutável (não pode alterar depois de criada)

# Usa parênteses → ( )

# Mais usada para dados fixos



# DICIONÁRIO (dict)

# Armazena pares chave: valor

# É mutável

# Usa chaves → { }

# Exemplo: {"nome": "Ana", "idade": 25}

# Acesso pelos nomes das chaves → dict["nome"]


# DICIONÁRIO (dict)

# Armazena pares chave: valor

# É mutável

# Usa chaves → { }

# Exemplo: {"nome": "Ana", "idade": 25}

# Acesso pelos nomes das chaves → dict["nome"]
